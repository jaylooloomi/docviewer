            content = el.textContent.trim();

          // ignore selector errors
        }

        // If not available from DOM, try parsedData contract_chunks (defensive)
        if (!content && parsedData && parsedData.return_content && parsedData.return_content.contract_chunks && parsedData.return_content.contract_chunks[id]) {
          content = String(parsedData.return_content.contract_chunks[id].content || '').trim();
        }

        // Limit size for messages
        if (content && content.length > 8000) content = content.slice(0, 8000);

        chunks[id] = { content };
      });
    }

    frame.contentWindow.postMessage(
      {
        type: 'HIGHLIGHT_BLOCKS',
        blockIds: ids,
        result: result,
        chunks: chunks,
      },
      '*'
